🐍 Python: Шпаргалка для новичков

# Числа
age = 25              # int
price = 99.9          # float

# Строки
name = "Анна"
text = 'Привет'

# Булевы значения
is_student = True
is_work = False

# Ничего
nothing = None

📋 Базовые операции

# Арифметика
5 + 3      # 8
5 - 3      # 2
5 * 3      # 15
5 / 2      # 2.5
5 // 2     # 2 (целочисленное)
5 % 2      # 1 (остаток)
5 ** 2     # 25 (степень)

# Со строками
"При" + "вет"        # "Привет"
"ха" * 3             # "хахаха"
len("Привет")        # 6 (длина)

🎯 Условные операторы

if age >= 18:
    print("Взрослый")
elif age >= 12:
    print("Подросток")
else:
    print("Ребёнок")

# Короткая запись (тернарный оператор)
status = "Взрослый" if age >= 18 else "Не взрослый"

🔄 Циклы

# for: перебор последовательности
for i in range(5):     # 0,1,2,3,4
    print(i)

for char in "Привет":
    print(char)        # П р и в е т

# while: пока условие истинно
i = 0
while i < 5:
    print(i)
    i += 1

📚 Коллекции

# Список (изменяемый)
fruits = ["яблоко", "банан", "апельсин"]
fruits[0]          # "яблоко"
fruits.append("киви")     # добавить
fruits.remove("банан")    # удалить
len(fruits)        # длина

# Кортеж (неизменяемый)
colors = ("красный", "зеленый", "синий")
colors[0]          # "красный"

# Словарь (ключ → значение)
person = {"name": "Анна", "age": 25}
person["name"]          # "Анна"
person["city"] = "Москва"  # добавить

# Множество (уникальные значения)
unique = {1, 2, 3, 3, 3}
# {1, 2, 3}

🔧 Функции

# Простая функция
def greet(name):
    return f"Привет, {name}!"

print(greet("Анна"))  # "Привет, Анна!"

# С параметром по умолчанию
def power(x, n=2):
    return x ** n

power(3)      # 9 (3²)
power(3, 3)   # 27 (3³)

# Лямбда-функция (однострочная)
square = lambda x: x ** 2
square(5)     # 25

📂 Работа со строками

text = "  Привет, Мир!  "

text.lower()        # "  привет, мир!  "
text.upper()        # "  ПРИВЕТ, МИР!  "
text.strip()        # "Привет, Мир!" (убрать пробелы)
text.replace("Мир", "Python")  # "Привет, Python!"
text.split(",")     # ["Привет", " Мир! "]

🧮 Полезные функции

print("hello")      # вывести в консоль
input("Введите имя: ")  # получить ввод

type(5)            # <class 'int'>
int("123")         # 123 (строка → число)
str(123)           # "123" (число → строка)

len([1, 2, 3])     # 3 (длина)
sum([1, 2, 3])      # 6 (сумма)
max([1, 5, 3])     # 5
min([1, 5, 3])     # 1

💡 Ошибки и отладка

try:
    number = int(input("Введите число: "))
    print(10 / number)
except ValueError:
    print("Это не число!")
except ZeroDivisionError:
    print("На ноль делить нельзя!")
else:
    print("Всё ок!")
finally:
    print("Конец программы")

📁 Файлы

# Запись в файл
with open("file.txt", "w") as f:
    f.write("Привет, файл!")

# Чтение из файла
with open("file.txt", "r") as f:
    content = f.read()  # весь файл
    lines = f.readlines()  # список строк

🎯 Срезы

text = "Python"
text[0]      # "P"
text[-1]     # "n" (последний)
text[1:4]    # "yth" (с 1 по 3)
text[:3]     # "Pyt" (с начала до 2)
text[3:]     # "hon" (с 3 до конца)
text[::-1]   # "nohtyP" (развернуть)

🧩 Генераторы списков

# Квадраты чисел от 0 до 9
squares = [x**2 for x in range(10)]
# [0, 1, 4, 9, 16, 25, 36, 49, 64, 81]

# Только чётные
evens = [x for x in range(10) if x % 2 == 0]
# [0, 2, 4, 6, 8]

🚀 Советы новичкам

  1. Отступы важны! Python использует отступы вместо {}
  2. Имена переменных: только буквы, цифры и _, не начинать с цифры
  3. Комментарии: # это комментарий
  4. Помощь: help(print) или dir(str)
  5. Главная боль: забыл : после if, for, def 😅
